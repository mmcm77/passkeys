Integration Example

<!-- 1. Include the PayAuth SDK -->
<script src="https://your-domain.com/sdk/payauth.min.js"></script>

<!-- 2. Create a container for the button -->
<div id="passkey-container"></div>

<!-- 3. Initialize the SDK -->
<script>
  const payAuth = PayAuth.init({
    merchantId: 'merchant_12345',
    container: '#passkey-container',
    theme: 'light',
    onSuccess: (userData) => {
      console.log('User authenticated:', userData.email);
      // Process payment or complete checkout here
    },
    onCancel: () => {
      console.log('Authentication cancelled');
    },
    onError: (error) => {
      console.error('Authentication error:', error);
    }
  });
</script>
